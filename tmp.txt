void selectionSort(int arr[], int n)
{
	int counter1;
	int counter2;
	int tmp;
	int min;

	counter1 = 0;
	while(counter1 < n - 1)
	{
		min = counter1;
		counter2 = counter1 + 1;
		while(counter2 < n)
		{
			if (arr[counter2] < arr[min])
				min = counter2;
			counter2 ++;
		}
		tmp = arr[counter1];
		arr[counter1] = arr[min];
		arr[min] = tmp;
		counter1 ++;
	}
}
